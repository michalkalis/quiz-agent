SELECT datname FROM pg_database WHERE datistemplate=false ORDER BY 1;
-- then \c <that_db> and:
SELECT extname||' '||extversion FROM pg_extension ORDER BY extname;                                  -- expect 'vector ...'
SELECT column_name, data_type, udt_name FROM information_schema.columns
  WHERE table_name='questions' AND column_name='embedding';                                          -- expect udt_name=vector
SELECT indexname, indexdef FROM pg_indexes WHERE tablename='questions';                              -- look for ivfflat/hnsw vector index
SELECT count(*) FROM questions;                                                                       -- Q1 total

SELECT language, count(*) FROM questions GROUP BY language ORDER BY 2 DESC;
SELECT id FROM questions WHERE embedding IS NOT NULL
  ORDER BY embedding <=> (SELECT embedding FROM questions WHERE embedding IS NOT NULL LIMIT 1) LIMIT 1; -- proves <=> operator on real data, no OpenAI
